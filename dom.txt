  <script src="./jquery-3.7.1.min.js"></script>
  <script>
    $(document).ready(function () {
      $("#bt1").click(function () {
        alert($("#test1").text());

        $("#test1").text("I am added by Dom Manipulation");
        $("#test1").before("I am added by Dom Manipulation Using Before");
      });

      $("#bt2").click(function () {
        alert($("#test2").html());

        $("#test2").text("<h1>I am adding a HTML using jQuery & DOM </h1>");
        $("#test2").after(
          "<h1>I am adding a HTML After using jQuery & DOM </h1>"
        );
      });

      $("#bt3").click(function () {
        alert($("#test3").val());
        $("#test3").val("Added value using DOM");
      });

      $("#bt4").click(function () {
        $("#Divone").remove();
      });
    });
  </script>
